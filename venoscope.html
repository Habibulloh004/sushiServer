            clipLimit=Config.CLAHE_CLIP_LIMIT,
            tileGridSize=Config.CLAHE_TILE_SIZE
        )
        enhanced = clahe.apply(gray)
        
        # –®–∞–≥ 2: Bilateral —Ñ–∏–ª—å—Ç—Ä
        bilateral = cv2.bilateralFilter(
            enhanced,
            Config.BILATERAL_D,
            Config.BILATERAL_SIGMA_COLOR,
            Config.BILATERAL_SIGMA_SPACE
        )
        
        # –®–∞–≥ 3: –£–≤–µ–ª–∏—á–µ–Ω–∏–µ —Ä–µ–∑–∫–æ—Å—Ç–∏
        kernel = np.array([[-1, -1, -1],
                          [-1,  9, -1],
                          [-1, -1, -1]])
        sharpened = cv2.filter2D(bilateral, -1, kernel)
        
        return sharpened

        –§–∏–ª—å—Ç—Ä –§—Ä–∞–Ω–≥–∏ –¥–ª—è –≤—ã–¥–µ–ª–µ–Ω–∏—è —Å–æ—Å—É–¥–∏—Å—Ç—ã—Ö —Å—Ç—Ä—É–∫—Ç—É—Ä
        –û–¥–∏–Ω –∏–∑ –ª—É—á—à–∏—Ö –º–µ—Ç–æ–¥–æ–≤ –¥–ª—è –≤—ã–¥–µ–ª–µ–Ω–∏—è –≤–µ–Ω
        """
        if len(image.shape) == 3:
            gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        else:
            gray = image.copy()
        
        gray = cv2.normalize(gray, None, 0, 255, cv2.NORM_MINMAX).astype(np.float64)
        filtered = np.zeros_like(gray)
        
        # –û–±—Ä–∞–±–æ—Ç–∫–∞ –Ω–∞ —Ä–∞–∑–Ω—ã—Ö –º–∞—Å—à—Ç–∞–±–∞—Ö
        for scale in scales:
            sigma = scale
            # –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –ø—Ä–æ–∏–∑–≤–æ–¥–Ω—ã—Ö –ì–∞—É—Å—Å–∞
            gxx = ndimage.gaussian_filter(gray, sigma, order=[2, 0])
            gyy = ndimage.gaussian_filter(gray, sigma, order=[0, 2])
            gxy = ndimage.gaussian_filter(gray, sigma, order=[1, 1])
            
            # –ê–Ω–∞–ª–∏–∑ —Å–æ–±—Å—Ç–≤–µ–Ω–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π –º–∞—Ç—Ä–∏—Ü—ã –ì–µ—Å—Å–µ
            # ... (–ø–æ–ª–Ω—ã–π –∫–æ–¥ –≤ —Ñ–∞–π–ª–µ)
        
        return cv2.normalize(filtered, None, 0, 255, cv2.NORM_MINMAX).astype(np.uint8)
    
    @staticmethod
    def retinex_enhancement(image, sigma=15):
        """
        Retinex –∞–ª–≥–æ—Ä–∏—Ç–º –¥–ª—è —É–ª—É—á—à–µ–Ω–∏—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
        –•–æ—Ä–æ—à –¥–ª—è –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏–∏ –æ—Å–≤–µ—â–µ–Ω–∏—è
        """
        if len(image.shape) == 3:
            gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        else:
            gray = image.copy()
        
        gray_float = gray.astype(np.float32) + 1.0
        blurred = cv2.GaussianBlur(gray_float, (0, 0), sigma)
        retinex = np.log(gray_float) - np.log(blurred)
        
        return cv2.normalize(retinex, None, 0, 255, cv2.NORM_MINMAX).astype(np.uint8)
    
    @staticmethod
    def ultimate_vein_enhancer(image):
        """
        –ö–æ–º–±–∏–Ω–∞—Ü–∏—è –≤—Å–µ—Ö –ª—É—á—à–∏—Ö –º–µ—Ç–æ–¥–æ–≤
        –î–ª—è –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –∫–∞—á–µ—Å—Ç–≤–∞ –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏–∏
        """
        # 1. –ö–æ—Ä—Ä–µ–∫—Ü–∏—è –æ—Å–≤–µ—â–µ–Ω–∏—è
        illumination_corrected = AdvancedVeinProcessor.homomorphic_filter(gray)
        
        # 2. –ú—É–ª—å—Ç–∏–º–∞—Å—à—Ç–∞–±–Ω–æ–µ —É–ª—É—á—à–µ–Ω–∏–µ
        multi_scale = AdvancedVeinProcessor.multi_scale_enhancement(illumination_corrected)
        
        # 3. –ê–¥–∞–ø—Ç–∏–≤–Ω–æ–µ —É–ª—É—á—à–µ–Ω–∏–µ
        adaptive = AdvancedVeinProcessor.adaptive_enhancement(multi_scale)
        
        # 4. Matched filter
        matched = AdvancedVeinProcessor.matched_filter(adaptive)
        
        # 5. –§–∏–Ω–∞–ª—å–Ω–∞—è CLAHE
        clahe = cv2.createCLAHE(clipLimit=4.0, tileGridSize=(8, 8))
        final = clahe.apply(matched)
        
        return final

class VeinAnalyzer:
    """–ê–Ω–∞–ª–∏–∑ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫ –≤–µ–Ω"""
    
    @staticmethod
    def measure_vein_width(binary_image):
        """–ò–∑–º–µ—Ä–µ–Ω–∏–µ —à–∏—Ä–∏–Ω—ã –≤–µ–Ω"""
        from skimage import morphology
        skeleton = morphology.skeletonize(binary_image > 0)
        distance = ndimage.distance_transform_edt(binary_image)
        widths = distance[skeleton] * 2
        
        return {
            'mean_width': np.mean(widths),
            'max_width': np.max(widths),
            'min_width': np.min(widths)
        }
